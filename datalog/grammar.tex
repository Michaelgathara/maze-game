\begin{grammar}
<toplvl-rule> ::= <rule> | <hclause> 

<rule> ::= "[" <hd-item>* "<--" <bd-item>* "]"
      \alt "[" <bd-item>* "-->" <hd-item>* "]" 

<bd-item> ::= <rule> | <bclause> 

<hd-item> ::= <rule> | <hclause> 

<bclause> ::= "("<tag> <ibclause>*")"  
         \alt "(=" <var> "("<tag> <ibclause>*"))"

<hclause> ::= "("<tag> <ihclause>*")" 
         \alt "(=" <var> "("<tag> <ihclause>*"))"

<atom> ::= <var> | <lit>

<lit> ::= <string> | <number>

% \end{grammar}
